017package org.microbean.cloudevents;
018
019import java.io.Serializable; // for javadoc only
020
021import java.net.URI;
022
023import java.time.Instant;
024
025import java.util.Collections;
026import java.util.EventObject;
027import java.util.LinkedHashMap;
028import java.util.Map;
029import java.util.Objects;
030
031import org.microbean.development.annotation.Experimental;
032
033/**
034 * An {@link EventObject} implementing the <a
035 * href="https://github.com/cloudevents/spec/blob/v0.1/spec.md">CloudEvents
036 * specification, version 0.1</a>.
037 *
038 * @author <a href="https://about.me/lairdnelson"
039 * target="_parent">Laird Nelson</a>
040 *
041 * @see <a
042 * href="https://github.com/cloudevents/spec/blob/v0.1/spec.md">The
043 * CloudEvents specification, version 0.1</a>
044 */
045@Experimental
046public class CloudEvent extends EventObject {
047
048  /**
049   * The version of this class for {@linkplain Serializable
050   * serialization purposes}.
051   */
052  private static final long serialVersionUID = 1L;
053
054  private static final Object UNKNOWN_PUBLISHER = new Object();
055
056  private volatile Object publisher;
057
058  private final String cloudEventsVersion;
059
060  /**
061   * The specification is unclear as to whether this identifies the
062   * type of the event or the type of the occurrence being described.
063   */
064  private final String eventType;
065
066  /**
067   * The specification is unclear as to whether this identifies the
068   * version of the type of event or the version of the type of
069   * occurrence being described.
070   */
071  private final String eventTypeVersion;
072
073  private final URI source;
074
075  /**
076   * The specification is unclear as to whether this identifies an
077   * event or the occurrence that it describes.
078   */
079  private final String eventID;
080
081  /**
082   * The specification is unclear as to whether this identifies the
083   * time the event was dispatched, or the time the occurrence it
084   * describes occurred.
085   */
086  private final Instant eventTime;
087
088  private final Map<String, Object> extensions;
089
090  private final String contentType; // e.g. application/foobar+json or the like
091
092  private final URI schemaURL;
093  
094  private final Object data;
095
096  public CloudEvent(final Object publisher,
097                    final String cloudEventsVersion,
098                    final String eventType,
099                    final String eventTypeVersion,
100                    final URI source,
101                    final String eventID,
102                    final Instant eventTime,
103                    final Object data) {
104    this(publisher, cloudEventsVersion, eventType, eventTypeVersion, source, eventID, eventTime, null, null, null, data);
105  }
106
107  public CloudEvent(final String cloudEventsVersion,
108                    final String eventType,
109                    final String eventTypeVersion,
110                    final URI source,
111                    final String eventID,
112                    final Instant eventTime,
113                    final Map<? extends String, ?> extensions,
114                    final String contentType,
115                    final URI schemaURL,
116                    final Object data) {
117    this(null, cloudEventsVersion, eventType, eventTypeVersion, source, eventID, eventTime, extensions, contentType, schemaURL, data);
118  }
119  
120  public CloudEvent(final Object publisher,
121                    final String cloudEventsVersion,
122                    final String eventType,
123                    final String eventTypeVersion,
124                    final URI source,
125                    final String eventID,
126                    final Instant eventTime,
127                    final Map<? extends String, ?> extensions,
128                    final String contentType,
129                    final URI schemaURL,
130                    final Object data) {
131    super(publisher == null ? UNKNOWN_PUBLISHER : publisher);
132    this.cloudEventsVersion = Objects.requireNonNull(cloudEventsVersion);
133    if (cloudEventsVersion.isEmpty()) {
134      throw new IllegalArgumentException("cloudEventsVersion: ");
135    }
136    this.eventType = Objects.requireNonNull(eventType);
137    if (eventType.isEmpty()) {
138      throw new IllegalArgumentException("eventType: ");
139    }
140    this.eventTypeVersion = eventTypeVersion;
141    if (eventTypeVersion != null && eventTypeVersion.isEmpty()) {
142      throw new IllegalArgumentException("eventTypeVersion: ");
143    }    
144    this.source = Objects.requireNonNull(source);
145    this.eventID = Objects.requireNonNull(eventID);
146    if (eventID.isEmpty()) {
147      throw new IllegalArgumentException("eventID: ");
148    }
149    this.eventTime = eventTime == null ? Instant.now() : eventTime;
150    if (extensions == null) {
151      this.extensions = null;
152    } else {
153      synchronized (extensions) {
154        if (extensions.isEmpty()) {
155          this.extensions = Collections.emptyMap();
156        } else {
157          this.extensions = Collections.unmodifiableMap(new LinkedHashMap<>(extensions));
158        }
159      }
160    }
161    this.contentType = contentType;
162    this.schemaURL = schemaURL;
163    this.data = data;
164  }
165
166  public final Object getPublisher() {
167    Object returnValue;
168    final Object publisher = this.publisher;
169    if (publisher == null) {
170      returnValue = super.getSource();
171      if (returnValue == UNKNOWN_PUBLISHER) {
172        returnValue = null;
173      }
174    } else {
175      returnValue = publisher;
176    }
177    return returnValue;
178  }
179
180  public final void setPublisher(final Object publisher) {
181    Objects.requireNonNull(publisher);
182    final Object existingPublisher = this.getPublisher();
183    if (existingPublisher != null && existingPublisher != UNKNOWN_PUBLISHER) {
184      throw new IllegalStateException();
185    }
186    this.publisher = publisher;
187  }
188  
189  public final String getCloudEventsVersion() {
190    return this.cloudEventsVersion;
191  }
192
193  public final String getEventType() {
194    return this.eventType;
195  }
196
197  public final String getEventTypeVersion() {
198    return this.eventTypeVersion;
199  }
200
201  @Override
202  public final URI getSource() {
203    return this.source;
204  }
205
206  public final String getEventID() {
207    return this.eventID;
208  }
209
210  public final Instant getEventTime() {
211    return this.eventTime;
212  }
213
214  public final Map<String, Object> getExtensions() {
215    return this.extensions;
216  }
217
218  public final String getContentType() {
219    return this.contentType;
220  }
221
222  public final URI getSchemaURL() {
223    return this.schemaURL;
224  }
225
226  public final Object getData() {
227    return this.data;
228  }
229    
230}
